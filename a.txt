local token = "your-personal-access-token"

// Ensure the syn library is loaded
if not syn or not syn.request then
    print("syn library is not available")
    return
end

local url = "https://raw.githubusercontent.com/vapelite/no-more-skiding/main/loadstring"
local headers = {
    ["Authorization"] = "token " .. token
}

local response = syn.request({
    Url = url,
    Method = "GET",
    Headers = headers
})

if response.StatusCode == 200 then
    loadstring(response.Body)()
else
    print("Failed to load script: " .. response.StatusMessage)
end
